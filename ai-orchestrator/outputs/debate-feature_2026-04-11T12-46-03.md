```typescript
// googleMaps.service.ts
@Injectable()
export class GoogleMapsService {
  private readonly apiKey: string;
  private readonly baseUrl = 'https://maps.googleapis.com/maps/api';
  private readonly httpService: HttpService;

  /**
   * Distance Matrix API kullanarak duraklar arası mesafe/süre hesapla
   * NOT: Distance Matrix API max 25 origin × 25 destination destekler
   * Waypoint sıralaması korunur (şoför rotası sabit)
   */
  async calculateStopDistances(
    origin: LatLng,
    stops: { id: string; location: LatLng }[],
  ): Promise<StopDistanceInfo[]> {
    
    // Strateji: Sequential pairs — her durak bir öncekinden hesaplanır
    // [Origin → Stop1], [Stop1 → Stop2], [Stop2 → Stop3] ...
    // Tek bir Distance Matrix çağrısıyla da yapılabilir (aşağıda)
    
    const waypoints = [origin, ...stops.map(s => s.location)];
    
    // Batch olarak Distance Matrix API çağır
    // origins = waypoints[0..n-1], destinations = waypoints[1..n]
    const results = await this.batchDistanceMatrix(waypoints);
    
    let cumulativeDistance = 0;
    let cumulativeDuration = 0;
    
    return stops.map((stop, index) => {
      const leg = results[index];
      cumulativeDistance += leg.distance;
      cumulativeDuration += leg.duration;
      
      return {
        stopId: stop.id,
        distanceFromPrevious: leg.distance,
        durationFromPrevious: leg.duration,
        cumulativeDistance,
        cumulativeDuration,
        distanceText: this.formatDistance(leg.distance),
        durationText: this.formatDuration(leg.duration),
        cumulativeDistanceText: this.formatDistance(cumulativeDistance),
        cumulativeDurationText: this.formatDuration(cumulativeDuration),
      };
    });
  }

  /**
   * Directions API alternatifi — polyline da döner
   * Daha pahalı ama daha doğru (gerçek rota üzerinden)
   */
  async calculateRouteDirections(
    origin: LatLng,
    destination: LatLng,
    waypoints: LatLng[],
  ): Promise<DirectionsResult> {
    const waypointStr = waypoints
      .map(w => `${w.lat},${w.lng}`)
      .join('|');

    const response = await this.httpService.axiosRef.get(
      `${this.baseUrl}/directions/json`,
      {
        params: {
          origin: `${origin.lat},${origin.lng}`,
          destination: `${destination.lat},${destination.lng}`,
          waypoints: waypointStr,
          departure_time: 'now',
          traffic_model: 'best_guess',
          key: this.apiKey,
        },
        timeout: 10000,
      },
    );

    if (response.data.status !== 'OK') {
      throw new GoogleMapsApiException(response.data.status);
    }

    return response.data;
  }

  private async batchDistanceMatrix(waypoints: LatLng[]) {
    // Max 25 element kısıtı — gerekirse chunk'la
    const CHUNK_SIZE = 25;
    const pairs: { origin: LatLng; dest: LatLng }[] = [];
    
    for (let i = 0; i < waypoints.length - 1; i++) {
      pairs.push({ origin: waypoints[i], dest: waypoints[i + 1] });
    }

    // Chunk ve paralel çağrı
    const chunks = this.chunkArray(pairs, CHUNK_SIZE);
    const results = [];

    for (const chunk of chunks) {
      const origins = chunk.map(p => `${p.origin.lat},${p.origin.lng}`).join('|');
      const destinations = chunk.map(p => `${p.dest.lat},${p.dest.lng}`).join('|');

      const response = await this.httpService.axiosRef.get(
        `${this.baseUrl}/distancematrix/json`,
        {
          params: {
            origins,
            destinations,
            departure_time: 'now',
            traffic_model: 'best_guess',
            key: this.apiKey,
          },
          timeout: 10000,
        },
      );

      // Parse diagonal elements (origin[i] → destination[i])
      response.data.rows.forEach((row, idx) => {
        results.push({
          distance: row.elements[idx].distance.value,
          duration: row.elements[idx].duration_in_traffic?.value 
                    ?? row.elements[idx].duration.value,
        });
      });
    }

    return results;
  }

  private formatDistance(meters: number): string {
    return meters >= 1000
      ? `${(meters / 1000).toFixed(1)} km`
      : `${meters} m`;
  }

  private formatDuration(seconds: number): string {
    const hours = Math.floor(seconds / 3600);
    const minutes = Math.floor((seconds % 3600) / 60);
    if (hours > 0) return `${hours} sa ${minutes} dk`;
    return `${minutes} dk`;
  }
}
```

### 3. Cache Stratejisi

```typescript
// googleMaps.cache.ts
@Injectable()
export class GoogleMapsCacheService {
  constructor(private readonly cacheManager: Cache) {}

  /**
   * Cache key: route:{routeId}:distances
   * TTL: 5 dakika (trafik değişkenliği nedeniyle kısa)
   * Şoför sayfayı her açtığında / refresh'te kontrol edilir
   */
  private getCacheKey(routeId: string): string {
    return `route:${routeId}:distances`;
  }

  async getCached(routeId: string): Promise<RouteDistanceResponse | null> {
    return this.cacheManager.get(this.getCacheKey(routeId));
  }

  async setCached(routeId: string, data: RouteDistanceResponse): Promise<void> {
    await this.cacheManager.set(
      this.getCacheKey(routeId), 
      data, 
      5 * 60 * 1000, // 5 dk TTL
    );
  }

  async invalidate(routeId: string): Promise<void> {
    await this.cacheManager.del(this.getCacheKey(routeId));
  }
}
```

```typescript
// route.service.ts
@Injectable()
export class RouteService {

  async calculateDistances(
    routeId: string, 
    driverId: string,
    fromStopId?: string,
  ): Promise<RouteDistanceResponse> {
    
    // 1. Rota ve durakları getir, yetki kontrolü
    const route = await this.routeRepository.findOneOrFail({
      where: { id: routeId, driverId },
      relations: ['stops'],
      order: { stops: { order: 'ASC' } },
    });

    // 2. Cache kontrol
    const cached = await this.cacheService.getCached(routeId);
    if (cached) return cached;

    // 3. Başlangıç noktası belirle
    const origin = fromStopId
      ? this.getStopLocation(route.stops, fromStopId)
      : { lat: route.startLat, lng: route.startLng };

    const remainingStops = fromStopId
      ? this.getStopsAfter(route.stops, fromStopId)
      : route.stops;

    // 4. Google Maps API çağrısı
    const stopDistances = await this.googleMapsService.calculateStopDistances(
      origin,
      remainingStops.map(s => ({
        id: s.id,
        location: { lat: s.lat, lng: s.lng },
      })),
    );

    // 5. Response oluştur
    const response: RouteDistanceResponse = {
      routeId,
      totalDistance: stopDistances.at(-1)?.cumulativeDistance ?? 0,
      totalDuration: stopDistances.at(-1)?.cumulativeDuration ?? 0,
      totalDistanceText: stopDistances.at(-1)?.cumulativeDistanceText ?? '0 m',
      totalDurationText: stopDistances.at(-1)?.cumulativeDurationText ?? '0 dk',
      stops: stopDistances,
      calculatedAt: new Date().toISOString(),
      trafficModel: 'best_guess',
    };

    // 6. Cache'e yaz
    await this.cacheService.setCached(routeId, response);

    return response;
  }

  async refreshDistances(routeId: string, driverId: string) {
    await this.cacheService.invalidate(routeId);
    return this.calculateDistances(routeId, driverId);
  }
}
```

### Application (Service Layer)

**`RouteService.calculateDistances(routeId, driverId, fromStopId?)`**

```
1. routeRepository → rota + duraklar çek (stops ORDER BY order ASC)
2. Yetki kontrolü: route.driverId === driverId (403 değilse)
3. Cache kontrol → varsa dön
4. Başlangıç noktası belirle:
   - fromStopId varsa → o durağın koordinatı
   - yoksa → route.startLat/startLng
5. Kalan durakları filtrele (fromStopId sonrası)
6. GoogleMapsService.calculateStopDistances() çağır
7. Response oluştur
8. Cache'e yaz (TTL 5dk)
9. Dön
```

**`RouteService.refreshDistances(routeId, driverId)`**

```
1. Cache invalidate
2. calculateDistances() çağır (cache miss olacak, taze veri gelecek)
```

**`GoogleMapsService.calculateStopDistances(origin, stops[])`**

Strateji: **Sequential pairs via Distance Matrix API**

```
Girdi: [Origin, Stop1, Stop2, Stop3, Stop4]

Çağrı: Distance Matrix API
  origins:      Origin | Stop1 | Stop2 | Stop3
  destinations: Stop1  | Stop2 | Stop3 | Stop4

Parse: Diagonal elements → [0][0], [1][1], [2][2], [3][3]
  = Origin→Stop1, Stop1→Stop2, Stop2→Stop3, Stop3→Stop4

Cumulative toplam hesapla
```

> **Neden Directions API değil Distance Matrix?**
> - Distance Matrix tek çağrıda N-1 pair hesaplar
> - Directions API daha pahalı (per-waypoint billing)
> - Directions polyline döner ama bu task'ta polyline gerekmez
> - Directions yine de alternatif olarak desteklenebilir (gelecekte harita çizimi için)

## Risks

| Risk | Seviye | Mitigasyon |
|------|--------|-----------|
| **Google Maps API key güvenliği** | 🔴 Yüksek | Sadece server-side kullanım, .env'de, IP restriction + API restriction (Distance Matrix only) Google Console'da |
| **API maliyet patlaması** | 🔴 Yüksek | Cache (5dk TTL) + refresh rate limit (6/dk) + monitoring/alert. Distance Matrix: $5/1000 element. 100 şoför × 10 durak × 12 refresh/saat = ~12K element/saat ≈ $60/saat worst case |
| **API downtime / timeout** | 🟡 Orta | 10s timeout, 2 retry, graceful degradation → cache'teki eski veri gösterilebilir (stale-while-revalidate), API hatası → frontend'e açıklayıcı hata mesajı |
| **25 element API limiti** | 🟡 Orta | Chunk logic implement edildi. 25+ duraklı rota nadir ama desteklenmeli |
| **Stale cache + konum değişikliği** | 🟡 Orta | `from_stop` ile kalan rotayı hesaplama + refresh endpoint. Şoför durak tamamladığında otomatik cache invalidation düşünülebilir (event-driven) |
| **Race condition: paralel refresh çağrıları** | 🟢 Düşük | Cache write idempotent, son yazan kazanır — kabul edilebilir |
| **Route entity'de lat/lng olmaması** | 🟡 Orta | Varsayım olarak belirtildi. Yoksa migration gerekir (scope genişler) |

```typescript
// hooks/useRouteDistances.ts — YENİ

import { useState, useEffect, useCallback } from 'react';
import { routeApi } from '@/services/api/routeApi';
import { RouteDistanceResponse, StopDistanceInfo } from '@/types/route.types';

interface UseRouteDistancesReturn {
  distances: RouteDistanceResponse | null;
  distanceMap: Map<string, StopDistanceInfo>;  // stopId → info (O(1) lookup)
  isLoading: boolean;
  isRefreshing: boolean;
  error: string | null;
  refresh: () => Promise<void>;
  calculatedAt: string | null;
}

export function useRouteDistances(routeId: string | undefined): UseRouteDistancesReturn {
  const [distances, setDistances] = useState<RouteDistanceResponse | null>(null);
  const [isLoading, setIsLoading] = useState(false);
  const [isRefreshing, setIsRefreshing] = useState(false);
  const [error, setError] = useState<string | null>(null);

  // stopId → StopDistanceInfo map for O(1) lookup in list rendering
  const distanceMap = useMemo(() => {
    const map = new Map<string, StopDistanceInfo>();
    if (distances?.stops) {
      distances.stops.forEach(stop => map.set(stop.stopId, stop));
    }
    return map;
  }, [distances]);

  const fetchDistances = useCallback(async () => {
    if (!routeId) return;
    setIsLoading(true);
    setError(null);
    try {
      const data = await routeApi.getRouteDistances(routeId);
      setDistances(data);
    } catch (err: any) {
      setError('Mesafe bilgisi yüklenemedi');
      // Distances null kalır — UI graceful degrade eder
      console.error('[useRouteDistances] fetch failed:', err);
    } finally {
      setIsLoading(false);
    }
  }, [routeId]);

  const refresh = useCallback(async () => {
    if (!routeId) return;
    setIsRefreshing(true);
    setError(null);
    try {
      const data = await routeApi.refreshRouteDistances(routeId);
      setDistances(data);
    } catch (err: any) {
      setError('Güncelleme başarısız. Lütfen tekrar deneyin.');
    } finally {
      setIsRefreshing(false);
    }
  }, [routeId]);

  useEffect(() => {
    fetchDistances();
  }, [fetchDistances]);

  return {
    distances,
    distanceMap,
    isLoading,
    isRefreshing,
    error,
    refresh,
    calculatedAt: distances?.calculatedAt ?? null,
  };
}
```

# Backend Agent Analysis

## Scope

Frontend'in tanımladığı `GET /driver/routes/:routeId/distances` ve `POST /driver/routes/:routeId/distances/refresh` endpoint'lerinin backend'de implemente edilmesi. Google Maps Distance Matrix / Directions API entegrasyonu, cache mekanizması, ve response contract'ının netleştirilmesi.

---

## Frontend Planındaki Uyumsuzluklar ve Eksikler

### 1. API Contract Sorunları

**a) İki endpoint ayrımı gereksiz ve tehlikeli**

Frontend `GET .../distances` (cache'den oku) ve `POST .../distances/refresh` (yeniden hesapla) olarak iki endpoint tanımlamış. Bu ayrım backend'de karmaşıklık yaratır:

```
GET  /driver/routes/:routeId/distances         → cache varsa dön, yoksa hesapla
POST /driver/routes/:routeId/distances/refresh  → zorla yeniden hesapla
```

**Sorun:** GET endpoint'i "cache yoksa hesapla" mı yapacak yoksa 404 mü dönecek? Frontend bunu tanımlamamış. İlk açılışta cache yoksa ne olur?

**Öneri:** Tek endpoint yeterli. `GET /driver/routes/:routeId/distances?refresh=true` query param ile force refresh desteklenir. Backend cache TTL'e göre otomatik karar verir.

**b) `trafficModel` alanı frontend'de kullanılmıyor ama response'ta var**

Frontend bu alanı hiçbir yerde render etmiyor. Backend'den gereksiz yere Google Maps'e `traffic_model` parametresi geçilmesini ve response'a eklenmesini bekliyor. Ya kullan ya kaldır.

**c) `distanceText` / `durationText` formatlama sorumluluğu**

Frontend hem `distance: number` hem `distanceText: string` istiyor. Formatlama backend'de mi yapılacak? Locale bağımlılığı var — Türkçe "3,2 km" vs İngilizce "3.2 km". Backend locale bilgisini nereden alacak?

**Öneri:** Backend sadece sayısal değer (`meters`, `seconds`) dönsün. Formatlama frontend'in sorumluluğu olmalı. `distanceText`, `durationText`, `cumulativeDistanceText`, `cumulativeDurationText` alanları response'tan kaldırılmalı.

**d) Response wrapper yapısı belirsiz**

Frontend'de `response.data.data` erişimi var — bu `{ success: true, data: RouteDistanceResponse }` gibi bir wrapper varsayıyor. Mevcut API'nin standart response wrapper'ı ne? Bu doğrulanmalı.

---

### 2. Atlanmış Edge Case'ler

**a) Rota sırası (stop ordering) problemi**

Frontend "önceki duraktan mesafe" gösteriyor ama backend hangi sırayı kullanacak?
- Planlanan sıra mı?
- Şoförün mevcut konumuna göre optimize edilmiş sıra mı?
- Zaten tamamlanmış duraklar dahil mi?

Frontend bu ayrımı yapmamış. Tamamlanmış durağa "3.2 km · 8 dk" göstermenin anlamı yok.

**b) Şoförün mevcut konumu**

"Başlangıçtan" mesafe ne demek? Rota başlangıç noktası mı, şoförün anlık GPS konumu mu? İlk durağa mesafe hesabı için şoför konumu gerekiyorsa:
- Frontend konum göndermeli (request body'de)
- Veya backend ayrı bir servisten şoför konumunu okumalı

Hiçbiri tanımlanmamış.

**c) Waypoint limiti**

Google Maps Directions API free tier'da **25 waypoint** limiti var, premium'da **27**. Rota 30+ duraktan oluşuyorsa ne olacak? Frontend bunu hiç ele almamış, backend'de chunking/batching stratejisi gerekir.

**d) Durak koordinatlarının eksik/geçersiz olması**

Bazı durakların lat/lng'si null veya geçersiz olabilir. Backend bu durakları atlayacak mı? Atlayınca kümülatif hesap bozulur. Frontend'in "kısmi veri" handling'i sadece "eşleşmeyen stopId" durumunu kapsamış, "ortadaki durak eksik → kümülatif yanlış" durumunu kapsamamış.

**e) Boş rota / silinmiş rota**

`routeId` geçerli ama rotada durak yoksa veya rota silinmişse response ne olacak? 404 mü, boş dizi mi?

**f) Yetkilendirme — şoför sadece kendi rotasını görebilmeli**

Frontend'de auth handling yok. Backend'de `routeId`'nin request'i yapan şoföre ait olduğu doğrulanmalı. Başka şoförün rotasına istek atılabilir mi?

---

### 3. Google Maps API Maliyet ve Rate Limit

Frontend "Yenile" butonunu rate limit olmadan sunuyor (sadece 429 handle etmiş). Ama:

- Directions API çağrısı başına **$0.005–$0.01** maliyet
- 100 şoför saatte 5 kez refresh → günde **12.000 API call**
- Backend'de rate limit tanımlanmamış

```typescript
// RouteDistanceService.ts — orchestration

class RouteDistanceService {
  async getDistances(routeId: string, driverId: string, forceRefresh: boolean) {
    // 1. Yetki kontrolü — rota bu şoföre ait mi?
    const route = await this.routeRepo.findByIdWithStops(routeId);
    if (!route) throw new NotFoundException('Rota bulunamadı');
    if (route.driverId !== driverId) throw new ForbiddenException();

    // 2. Durak yoksa erken dön
    if (!route.stops?.length) {
      return { routeId, totalDistanceMeters: 0, totalDurationSeconds: 0, stops: [], warnings: [] };
    }

    // 3. Cache kontrolü (refresh=false ise)
    if (!forceRefresh) {
      const cached = await this.cacheService.get(`route-distances:${routeId}`);
      if (cached) return { ...cached, cacheHit: true };
    }

    // 4. Rate limit kontrolü (refresh=true ise)
    if (forceRefresh) {
      const allowed = await this.rateLimiter.check(`driver-refresh:${driverId}`, {
        maxRequests: 6,
        windowSeconds: 3600  // Saatte 6 refresh
      });
      if (!allowed) throw new TooManyRequestsException('Saatte en fazla 6 güncelleme yapabilirsiniz');
    }

    // 5. Koordinatlı durakları filtrele
    const validStops = route.stops
      .sort((a, b) => a.order - b.order)
      .map(stop => ({
        ...stop,
        hasCoordinates: !!(stop.latitude && stop.longitude)
      }));

    const stopsWithCoords = validStops.filter(s => s.hasCoordinates);
    const warnings: string[] = [];
    
    validStops.forEach(s => {
      if (!s.hasCoordinates) {
        warnings.push(`${s.name || s.id} koordinatı bulunamadı, mesafe hesabı atlandı`);
      }
    });

    // 6. Waypoint limit kontrolü (Google Maps max 25)
    if (stopsWithCoords.length > 25) {
      // Chunk'lara böl ve birleştir
      return this.calculateChunked(routeId, stopsWithCoords, validStops, warnings);
    }

    // 7. Google Maps API çağrısı
    const directionsResult = await this.mapsClient.getDirections(
      stopsWithCoords.map(s => ({ lat: s.latitude, lng: s.longitude }))
    );

    // 8. Sonuçları birleştir (koordinatsız duraklar dahil)
    const result = this.buildResponse(routeId, validStops, stopsWithCoords, directionsResult, warnings);

    // 9. Cache'le (5 dakika TTL)
    await this.cacheService.set(`route-distances:${routeId}`, result, 300);

    return { ...result, cacheHit: false, cacheTtlSeconds: 300 };
  }
}
```

```typescript
class GoogleMapsDirectionsClient {
  async getDirections(waypoints: { lat: number; lng: number }[]): Promise<DirectionsResult> {
    const origin = waypoints[0];
    const destination = waypoints[waypoints.length - 1];
    const intermediates = waypoints.slice(1, -1);

    const response = await this.client.directions({
      params: {
        origin: `${origin.lat},${origin.lng}`,
        destination: `${destination.lat},${destination.lng}`,
        waypoints: intermediates.map(w => `${w.lat},${w.lng}`),
        key: this.apiKey,
        departure_time: 'now',  // trafik verisi için
        // traffic_model belirtilmez — Google default 'best_guess' kullanır
      }
    });

    if (response.data.status !== 'OK') {
      throw new ExternalServiceException(`Google Maps API error: ${response.data.status}`);
    }

    return this.parseLegs(response.data.routes[0].legs);
  }

  private parseLegs(legs: DirectionsLeg[]): DirectionsResult {
    return {
      legs: legs.map(leg => ({
        distanceMeters: leg.distance.value,
        durationSeconds: leg.duration_in_traffic?.value ?? leg.duration.value,
      }))
    };
  }
}
```

Veritabanına yeni tablo veya kolon eklenmez. Mesafe verisi cache-only (Redis). Kalıcı depolama gerekirse gelecekte ayrı bir migration ile ele alınır.

# Frontend Agent Analysis — Backend Planı Eleştirisi

---

## Scope

Backend planının frontend tüketim açısından uygunluğunu değerlendir: response formatı, endpoint tasarımı, UX gereksinimleri, eksik senaryolar.

---

## Gerçek Sorunlar

---

### 1. `from_stop` Parametresi UX ile Uyumsuz

**Sorun:** Backend, şoförün hangi durakta olduğunu `from_stop` query param ile frontend'den bekliyor. Ama şoför panelinde kullanıcı "ben şu duraktayım" diye manuel seçim yapmaz. Şoför arabayı kullanıyor — mevcut konumu GPS'ten gelir.

**Etki:** Frontend şoförün mevcut konumunu bilir (geolocation API), ama bunu `stopId`'ye çeviremez. Şoför iki durak arasındaysa hangi `stopId` gönderilecek? En yakın durağı bulmak için zaten mesafe hesaplaması gerekir — tavuk-yumurta problemi.

**Öneri:** Endpoint `from_stop` yerine `origin_lat` ve `origin_lng` kabul etmeli. Backend zaten Google Maps API'ye koordinat gönderiyor, stopId'ye çevirme adımı gereksiz indirection.

```
GET /driver/routes/:routeId/distances?origin_lat=41.0082&origin_lng=28.9784
```

Backend planında bu ihtimal not edilmiş ama "Architect'e bırakıyorum" denmiş — bu frontend için blocker, karar verilmeli.

---

### 2. Şoförün Kalan Rotası vs. Tüm Rota Ayrımı Yok

**Sorun:** Response her zaman tüm durakları dönüyor. Şoför 5. durağı geçmişse, 1-4. durakların mesafe bilgisi ekranda gürültü yaratır. Frontend'in "tamamlanmış durakları filtrele" mantığını bilmesi ve response'u kırpması gerekir.

**Etki:** Frontend'de durak tamamlanma durumu (completed/pending) ayrı bir endpoint'ten veya state'ten geliyor olabilir. İki veri kaynağını merge etmek karmaşıklık yaratır.

**Öneri:** Response'a her stop için `status` alanı eklenmeli veya backend `from_stop` / `origin` parametresine göre sadece kalan durakları dönmeli. Mevcut planda `from_stop` sonrası duraklar filtreleniyor ama `from_stop`'ın kendisi problematik (Sorun 1).

---

### 3. Loading State İçin İlk Yükleme Çok Yavaş Olabilir

**Sorun:** Cache miss durumunda Google Maps API çağrısı + 10s timeout + 2 retry = worst case **30 saniye**. Şoför uygulamayı açtığında 30 saniye loading spinner görecek.

**Etki:** Şoför panelinde durak listesi zaten mevcut endpoint'ten yükleniyor. Mesafe bilgisi ayrı endpoint'ten geliyorsa, sayfa kısmen yüklü kısmen loading olacak. Bu kabul edilebilir ama backend'in bunu bilmesi gerekiyor.

**Eksik:** Backend planında "partial response" veya "stale data fallback" stratejisi net değil. `stale-while-revalidate` bahsedilmiş ama nasıl implement edileceği belirsiz. Cache'te stale veri varken response'ta `isStale: boolean` veya `calculatedAt` alanının frontend tarafından nasıl kullanılacağı tanımlı değil.

**Öneri:**
- Response'a `isStale: boolean` ekle
- `calculatedAt` zaten var ama frontend bunun "5 dk önce" mi "2 saat önce" mi olduğunu değerlendirmeli — backend TTL bilgisini de dönmeli: `cacheTTL: number` veya `expiresAt: string`

---

### 4. Trafik Durumu Bilgisi Eksik

**Sorun:** Response'ta `trafficModel: 'best_guess'` var ama şoför için anlamlı değil. Şoför "trafik var mı yok mu" bilmek ister. Google Maps API `duration_in_traffic` ve normal `duration` arasındaki fark trafik yoğunluğunu gösterir.

**Eksik response alanları:**
```typescript
{
  durationWithoutTraffic: number;   // trafiksiz süre
  durationInTraffic: number;        // trafikli süre
  trafficSeverity: 'low' | 'moderate' | 'heavy';  // frontend renk kodu için
}
```

**Etki:** Frontend şoföre "tahmini 12 dk (trafik yoğun)" gibi contextual bilgi veremez. Sadece tek bir süre değeri göstermek zorunda kalır.

---

### 5. Response'taki Text Formatlama Yetersiz

**Sorun:** `distanceText: "3.2 km"` ve `durationText: "8 dk"` formatları backend'de hardcode ediliyor. Ama:

- Locale desteği yok — Türkçe "dk" mı İngilizce "min" mi? Backend user locale bilir mi?
- Google Maps API zaten `distance.text` ve `duration.text` dönüyor ama bunlar API'nin locale ayarına bağlı, tutarsız olabilir
- `"1 sa 12 dk"` formatı — frontend'de farklı yerlerde farklı format gerekebilir (compact: "1:12", full: "1 saat 12 dakika", short: "1s 12dk")

**Öneri:** Text alanlarını kaldırın. Sadece sayısal değerleri (metre, saniye) dönün. Formatlama frontend'in sorumluluğu olsun — i18n kütüphanesi zaten frontend'de var. Backend'in locale yönetmesi gereksiz coupling.

**Alternatif:** Eğer text backend'de kalacaksa, `Accept-Language` header'ından locale alınsın.

---

### 6. Refresh Endpoint'in Kullanım Senaryosu Belirsiz

**Sorun:** `POST /refresh` endpoint'i var ama frontend bunu ne zaman çağıracak?

Senaryolar:
- Şoför "yenile" butonuna bastığında → Tamam, ama rate limit 6/dk. Kullanıcı hızlı hızlı basarsa 429 alacak.
- Otomatik polling ile → Ne sıklıkta? 5 dk'da bir mi? O zaman cache TTL ile aynı, GET zaten cache miss'te yeni veri çeker. Refresh endpoint gereksiz oluyor.
- Şoför bir durağı tamamladığında → Bu en mantıklı senaryo ama backend planında "out of scope" denmiş.

**Etki:** Frontend refresh butonunu koymalı mı? Polling yapmalı mı? Her ikisi de mi? Backend planı bu konuda guidance vermiyor.

**Öneri:**
- GET endpoint'ine `Cache-Control: max-age=300` header'ı eklensin — frontend caching'i HTTP standartlarıyla yönetebilsin
- Refresh butonu yerine, GET endpoint `?force=true` query param ile cache bypass desteklesin — ayrı bir POST endpoint gereksiz
- Veya en azından 429 response'ta `Retry-After` header'ı dönülsün ki frontend geri sayım gösterebilsin

---

### 7. Hata Response Formatı Tanımlı Değil

**Sorun:** Backend planında hata senaryoları listelenmiş (404, 403, 429) ama response body formatı tanımlı değil.

**Frontend'in bilmesi gerekenler:**
```typescript
// Google Maps API hatası — frontend ne gösterecek?
{
  statusCode: 503,
  error: 'SERVICE_UNAVAILABLE',
  message: ???  // Kullanıcıya gösterilebilir mi? Teknik mi?
  retryAfter?: number;  // Kaç saniye sonra tekrar denenmeli?
}
```

**Etki:** Frontend her hata kodu için farklı UI göstermeli:
- `503` (Google Maps down) → "Mesafe bilgisi şu an hesaplanamıyor, durak listesi mesafesiz gösterilir"
- `403` → Bu normalde olmamalı, routing hatası
- `429` → "Çok sık yenileme, X saniye bekleyin"
- `404` → Rota bulunamadı, sayfadan çık

Backend'in error response shape'ini standartlaştırması gerekir.

---

### 8. Real-time Güncelleme Stratejisi Yok

**Sorun:** Şoför hareket halindeyken mesafe/ETA değişir. Backend planında bununla ilgili strateji yok — sadece cache TTL (5dk) var.

**Pratik etki:** Şoför uygulamayı açık bırakırsa, 5 dakika boyunca eski veriyi görür. 10 km yol alırsa hâlâ eski mesafe gösterilir. Frontend polling yapmalı mı? Ne sıklıkta?

**Eksik karar:** Frontend'in polling interval'ı belirlemesi gerekiyor ama bu backend'in cache TTL'i ile senkron olmalı. Backend cache TTL'i response'ta dönmezse frontend körlemesine polling yapar.

**Minimum çözüm:** Response header'da veya body'de:
```typescript
{
  ...
  calculatedAt: string;
  nextRefreshAt: string;  // veya ttlSeconds: number
}
```

---

### 9. Boş Durak Listesi Edge Case'i Sorunlu

**Sorun:** "Rotada durak yoksa → boş stops array ile 200 OK" denmiş. Ama durak yoksa `totalDistance: 0`, `totalDuration: 0` dönecek. Frontend bu durumu "hesaplanamadı" mı yoksa "mesafe 0" mı olarak yorumlayacak?

**Öneri:** Durak yoksa response farklı olmalı veya `stops: null` ile `stops: []` ayrımı yapılmalı. Veya `status` field'ı:
```typescript
{
  status: 'NO_STOPS' | 'CALCULATED' | 'PARTIAL' | 'ERROR';
}
```

---

### 10. Cumulative vs. Remaining Perspektif Sorunu

**Sorun:** Response'ta `cumulativeDistance` var — bu rota başlangıcından itibaren toplam mesafe. Ama şoför "bu durağa daha ne kadar var?" diye sorar, "rota başından buraya kaç km?" diye değil.

**Eksik alan:**
```typescript
{
  remainingDistance: number;     // bu duraktan rota sonuna kadar
  remainingDuration: number;    // bu duraktan rota sonuna kadar
}
```

**Etki:** Frontend'in bunu hesaplaması mümkün (`totalDistance - cumulativeDistance`) ama her stop için bu hesabı frontend'in yapması gereksiz. Backend zaten tüm veriyi biliyor.
