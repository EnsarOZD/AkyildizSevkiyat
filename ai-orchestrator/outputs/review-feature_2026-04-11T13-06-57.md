public class AuthorizationBehavior<TRequest, TResponse>
    : IPipelineBehavior<TRequest, TResponse>
    where TRequest : IRequest<TResponse>   // ← IRequest — hem Command hem Query
{
    public async Task<TResponse> Handle(
        TRequest request,
        RequestHandlerDelegate<TResponse> next,
        CancellationToken cancellationToken)
    {
        if (request is IRequireRoles requireRoles)
        {

                currentUser.Roles.Contains(role)))
            {
                throw new ForbiddenAccessException(
                    "Bu işlem için yetkiniz bulunmuyor.");
            }
        }

        return await next();
    }
}

### Bulgu #4: Şifre Validation Yok

**Domain:**

```csharp
// Domain/Shared/PasswordPolicy.cs (YENİ DOSYA)
namespace AkyildizSevkiyat.Domain.Shared;

public static class PasswordPolicy
{
    public const int MinLength = 8;
    public const int MaxLength = 128;
    public const bool RequireUppercase = true;
    public const bool RequireLowercase = true;
    public const bool RequireDigit = true;
    public const bool RequireSpecialChar = true;

    public static List<string> Validate(string? password)
    {
        var errors = new List<string>();

        if (string.IsNullOrWhiteSpace(password))
        {
            errors.Add("Şifre boş olamaz.");
            return errors; // Diğer kontroller anlamsız
        }

        if (password.Length < MinLength)
            errors.Add($"Şifre en az {MinLength} karakter olmalıdır.");

        if (password.Length > MaxLength)
            errors.Add($"Şifre en fazla {MaxLength} karakter olabilir.");

        if (RequireUppercase && !password.Any(char.IsUpper))
            errors.Add("En az bir büyük harf içermelidir.");

        if (RequireLowercase && !password.Any(char.IsLower))
            errors.Add("En az bir küçük harf içermelidir.");

        if (RequireDigit && !password.Any(char.IsDigit))
            errors.Add("En az bir rakam içermelidir.");

        if (RequireSpecialChar && password.All(char.IsLetterOrDigit))
            errors.Add("En az bir özel karakter içermelidir.");

        return errors;
    }
}
```

**Application:**

```csharp
// CreateUserCommandValidator.cs
public class CreateUserCommandValidator : AbstractValidator<CreateUserCommand>
{
    public CreateUserCommandValidator()
    {
        RuleFor(x => x.Username)
            .NotEmpty().WithMessage("Kullanıcı adı boş olamaz.")
            .MinimumLength(3);

        RuleFor(x => x.Password)
            .NotEmpty().WithMessage("Şifre boş olamaz.")
            .Custom((password, context) =>
            {
                var errors = PasswordPolicy.Validate(password);
                foreach (var error in errors)
                    context.AddFailure("Password", error);
            });
    }
}

// ResetPasswordCommandValidator.cs
public class ResetPasswordCommandValidator : AbstractValidator<ResetPasswordCommand>
{
    public ResetPasswordCommandValidator()
    {
        RuleFor(x => x.NewPassword)
            .NotEmpty().WithMessage("Yeni şifre boş olamaz.")
            .Custom((password, context) =>
            {
                var errors = PasswordPolicy.Validate(password);
                foreach (var error in errors)
                    context.AddFailure("NewPassword", error);
            });

        // Opsiyonel: eski şifre ile aynı olmamalı
        RuleFor(x => x)
            .Must(x => x.NewPassword != x.CurrentPassword)
            .WithMessage("Yeni şifre eski şifre ile aynı olamaz.")
            .When(x => !string.IsNullOrEmpty(x.CurrentPassword));
    }
}

**3. Composable — useDeliveryConfirmation.js**

```javascript
// composables/useDeliveryConfirmation.js

import { ref, onMounted, onUnmounted } from 'vue';
import api from '@/services/api';
import { useToast } from '@/composables/useToast';

const OFFLINE_QUEUE_KEY = 'pending_delivery_confirmations';

export function useDeliveryConfirmation(shipmentId) {
  const isSubmitting = ref(false);
  const lastError = ref(null);
  const isOffline = ref(!navigator.onLine);
  const pendingCount = ref(0);
  const toast = useToast();

  // Online/offline listener
  function handleOnline() {
    isOffline.value = false;
    flushOfflineQueue();
  }
  function handleOffline() {
    isOffline.value = true;
  }

  onMounted(() => {
    window.addEventListener('online', handleOnline);
    window.addEventListener('offline', handleOffline);
    pendingCount.value = getOfflineQueue().length;
  });

  onUnmounted(() => {
    window.removeEventListener('online', handleOnline);
    window.removeEventListener('offline', handleOffline);
  });

  async function confirmDelivery(payload) {
    if (isSubmitting.value) return; // double-click koruması
    
    isSubmitting.value = true;
    lastError.value = null;

    // Offline ise queue'ya ekle
    if (!navigator.onLine) {
      addToOfflineQueue({ shipmentId, payload, timestamp: Date.now() });
      toast.info('Bağlantı yok. Onay kaydedildi, internet gelince gönderilecek.');
      isSubmitting.value = false;
      return;
    }

    try {
      await api.post(`/api/shipments/${shipmentId}/confirm-delivery`, payload);
      toast.success('Teslimat onaylandı ✓');
      // navigate veya state güncelle
    } catch (error) {
      // Spesifik hata handling
      if (error.response?.status === 409) {
        lastError.value = 'Bu teslimat zaten onaylanmış.';
        toast.warning(lastError.value);
      } else if (error.response?.status === 400) {
        const msg = error.response.data?.message || 'Geçersiz istek.';
        lastError.value = msg;
        toast.error(msg);
      } else if (!error.response) {
        // Network hatası — offline queue'ya düş
        addToOfflineQueue({ shipmentId, payload, timestamp: Date.now() });
        lastError.value = 'Bağlantı kesildi. Onay kaydedildi.';
        toast.warning(lastError.value);
      } else {
        lastError.value = 'Beklenmeyen bir hata oluştu. Tekrar deneyin.';
        toast.error(lastError.value);
      }
    } finally {
      isSubmitting.value = false;
    }
  }

  // Basit localStorage offline queue
  function getOfflineQueue() {
    try {
      return JSON.parse(localStorage.getItem(OFFLINE_QUEUE_KEY) || '[]');
    } catch {
      return [];
    }
  }

  function addToOfflineQueue(item) {
    const queue = getOfflineQueue();
    queue.push(item);
    localStorage.setItem(OFFLINE_QUEUE_KEY, JSON.stringify(queue));
    pendingCount.value = queue.length;
  }

  async function flushOfflineQueue() {
    const queue = getOfflineQueue();
    if (queue.length === 0) return;

    const remaining = [];
    for (const item of queue) {
      try {
        await api.post(
          `/api/shipments/${item.shipmentId}/confirm-delivery`,
          item.payload
        );
        toast.success(`Bekleyen onay gönderildi (${item.shipmentId})`);
      } catch {
        remaining.push(item); // başarısızları tekrar kuyruğa al
      }
    }
    localStorage.setItem(OFFLINE_QUEUE_KEY, JSON.stringify(remaining));
    pendingCount.value = remaining.length;
  }

  return {
    isSubmitting,
    lastError,
    isOffline,
    pendingCount,
    confirmDelivery,
  };
}

// composables/usePasswordValidation.js

import { ref, computed, watch } from 'vue';

// Backend PasswordPolicy ile AYNI kurallar — tek kaynak olmalı
const RULES = [
  {
    id: 'minLength',
    label: 'En az 8 karakter',
    test: (pw) => pw.length >= 8,
  },
  {
    id: 'uppercase',
    label: 'En az bir büyük harf (A-Z)',
    test: (pw) => /[A-Z]/.test(pw),
  },
  {
    id: 'digit',
    label: 'En az bir rakam (0-9)',
    test: (pw) => /\d/.test(pw),
  },
  {
    id: 'special',
    label: 'En az bir özel karakter (!@#$...)',
    test: (pw) => /[^a-zA-Z0-9]/.test(pw),
  },
];

export function usePasswordValidation() {
  const password = ref('');
  const confirmPassword = ref('');

  const ruleResults = computed(() =>
    RULES.map((rule) => ({
      ...rule,
      passed: password.value ? rule.test(password.value) : false,
    }))
  );

  const strength = computed(() => {
    if (!password.value) return 'none';
    const passed = ruleResults.value.filter((r) => r.passed).length;
    if (passed <= 1) return 'weak';
    if (passed <= 2) return 'fair';
    if (passed <= 3) return 'good';
    return 'strong';
  });

  const strengthColor = computed(() => ({
    none: '#ccc',
    weak: '#e74c3c',
    fair: '#f39c12',
    good: '#3498db',
    strong: '#27ae60',
  }[strength.value]));

  const isValid = computed(() =>
    ruleResults.value.every((r) => r.passed)
  );

  const passwordsMatch = computed(() =>
    confirmPassword.value === password.value && password.value !== ''
  );

  const canSubmit = computed(() =>
    isValid.value && passwordsMatch.value
  );

  return {
    password,
    confirmPassword,
    ruleResults,
    strength,
    strengthColor,
    isValid,
    passwordsMatch,
    canSubmit,
  };
}

# Architect Agent — Akyıldız Sevkiyat Pre-Release Architectural Review

---

## Change

Akyıldız Sevkiyat uygulamasının yayın öncesi review'ında tespit edilen 14 bulgunun mimari uygunluk, katman sınır ihlalleri, entegrasyon riskleri ve uzun vadeli yapısal etkilerinin değerlendirmesi.

---

## Verdict

**CONDITIONAL APPROVAL — 6 Blokleyici Koşul ile**

Sistem mimari olarak Clean Architecture prensiplerine uygun kurgulanmış. Ancak uygulamada **3 yapısal mimari hata** ve **3 altyapı konfigürasyon açığı** mevcut. Bu 6 koşul kapatılmadan production deployment yapılmamalı.

---

## Reason

### 1. Authorization Mimarisi Kırık (Bulgu #2 + #3 — birlikte değerlendirilmeli)

Sistemde iki bağımsız auth katmanı olması gerekiyor:
- **Katman 1 (Transport):** ASP.NET Core authorization middleware → Controller/endpoint seviyesi
- **Katman 2 (Application):** MediatR pipeline `AuthorizationBehavior` → business operation seviyesi

**Mevcut durum:** Her iki katman da delik.

- Katman 1: 3 controller'da `[Authorize]` yok, global fallback policy yok. Transport seviyesinde sınır tamamen açık.
- Katman 2: `AuthorizationBehavior` generic constraint'i `ICommand<TResponse>` ile sınırlı. Tüm Query'ler pipeline auth'dan muaf.

**Mimari değerlendirme:** Bu iki bulgu ayrı bug değil, **tek bir mimari eksiklik**: defense-in-depth auth stratejisi tasarlanmamış. Backend agent'ın önerdiği çözüm doğru — her iki katmanı birlikte kapatmak gerekiyor. Ancak dikkat edilmesi gereken nokta: generic constraint değişikliği (`ICommand<T>` → `IRequest<T>`) tüm pipeline behavior'ların çalışma kapsamını genişletir. `ValidationBehavior`, `LoggingBehavior` gibi diğer behavior'lar varsa, bunların Query'lerle de uyumlu çalıştığı doğrulanmalı.

**Risk sınıfı:** Tam veri ifşası. Tüm iş verileri (sevkiyat, müşteri, stok, kullanıcı) anonim erişime açık.

---

### 2. Netsis Entegrasyon Sınırı Tanımsız (Bulgu #5 + #6 — birlikte değerlendirilmeli)

İki ayrı TODO, tek bir mimari sorun: **Netsis entegrasyon kontraktı netleştirilmemiş.**

- **Bulgu #5 (Stok Sync):** Gelen veri mapping'i eksik/belirsiz. Netsis'ten ne geldiği, nasıl dönüştürüleceği, hangi hata durumlarında ne yapılacağı tanımlı değil.
- **Bulgu #6 (CariKod Export):** Giden veri mapping'i eksik. İç müşteri ID → Netsis CariKod dönüşümü tasarlanmamış.

**Mimari değerlendirme:** Bunlar kod eksiği değil, **entegrasyon kontraktı eksikliği**. Anti-corruption layer (ACL) perspektifinden bakıldığında:

```
[Akyıldız Domain] ←→ [ACL / Adapter] ←→ [Netsis External System]
                          ↑
                    Bu katman tanımsız
```

Backend agent'ın önerdiği handler-level validation ve fallback doğru. Ancak mimari olarak bu mapping'lerin handler içinde inline olmaması, ayrı bir `INetsisMapper` veya `NetsisTranslationService` abstraction'ında toplanması gerekir. Bu yayın öncesi refactor gerektirmez — handler içinde doğru mapping ile gidilebilir — ama mapping mantığının tek noktada olması ikinci sprint'te yapılmalı.

**Kritik karar noktası — Bulgu #6 CariKod:** Eğer Customer entity'sinde `NetsisCariKod` alanı yoksa, bu bir **schema change** gerektirir. Migration, seed data, ve mevcut müşteri verilerinin CariKod ile eşleştirilmesi gerekir. Bu production'a gitmeden önce veri hazırlığı da dahil tamamlanmalı. CariKod'suz müşteriye export yapılması kesinlikle engellenmeli (hard fail, silent skip değil).

**Risk sınıfı:** Yanlış müşteriye fatura kesilmesi (yasal), yanlış stok verisi ile sevkiyat planlanması (operasyonel + finansal).

---

### 3. Secret Management Altyapı Kararı (Bulgu #1)

**Mimari değerlendirme:** Bu bir bug değil, **altyapı mimarisi kararı** eksikliği. Commit edilmiş secret'lar semptom — asıl sorun secret management stratejisinin belirlenmemiş olması.

Backend agent'ın önerdiği çözüm hiyerarşisi doğru:
- Development: `dotnet user-secrets`
- Production: Environment variables (minimum) veya vault (ideal)

**Ek mimari concern:** Secret'lar 3 farklı external sisteme ait (Google Maps, Netsis, ISS). Bu sistemlerin her birinin credential lifecycle'ı farklı. Rotation politikası ve credential sahibi (ops mı, dev mi?) netleştirilmeli. Bu yayın öncesi bir doküman gerektirmez ama rotation'ın fiilen yapılması gerekir.

**Risk sınıfı:** Tüm external sistem credential'larının ifşası. Supply chain boyutunda risk.

---

### 4. Pipeline Behavior Genişleme Etkisi (Bulgu #3'ten türeyen çapraz etki)

Backend agent `AuthorizationBehavior`'ın constraint'ini `ICommand<T>` → `IRequest<T>` olarak değiştirmeyi öneriyor. Bu doğru ve gerekli. Ancak mimari etki analizi:

**Kontrol edilmesi gerekenler:**
1. `ValidationBehavior<T>` — Query'ler için de çalışacak mı? Query'lerde FluentValidation validator yoksa sorun olmaz ama registration'da `typeof(IValidator<>)` resolve failure olmadığından emin olunmalı.
2. `LoggingBehavior<T>` — Query'leri de loglayacak mı? İsteniyorsa sorun yok, istemiyorsa performans etkisi minimal ama log noise artabilir.
3. `TransactionBehavior<T>` — Varsa, Query'leri transaction scope'a almak gereksiz ve zararlı. Bu behavior sadece `ICommand<T>` constraint'ini korumalı.

**Sonuç:** Behavior constraint değişikliği yapılırken, diğer pipeline behavior'ların constraint'leri ve kapsamları da gözden geçirilmeli. Bu 30 dakikalık bir kontrol ama atlanırsa runtime'da beklenmeyen transaction scope'lar veya validation hataları çıkabilir.

---

### 5. Offline-First Pattern Kararı (Bulgu #8 — Mimari etki)

Frontend agent'ın önerdiği localStorage-based offline queue doğru bir ilk adım. Ancak mimari olarak bu pattern **idempotency guarantee** gerektirir.

**Senaryo:**
```
Şoför → "Onayla" → Network fail → localStorage'a yazıldı → 
Network geldi → Retry → Ama ilk request aslında sunucuya ulaşmıştı, 
sadece response dönmemişti → Duplicate confirmation
```

**Gerekli backend desteği:**
- Teslimat onayı endpoint'i **idempotent** olmalı. Aynı `shipmentId` + `driverId` ile ikinci onay 409 dönmeli, ikinci bir teslimat kaydı oluşturmamalı.
- Frontend'in 409'u "başarılı" olarak treat etmesi gerekiyor (ki frontend agent bunu doğru handle etmiş).

**Bu cross-layer bir tasarım kararı:** Backend endpoint idempotent değilse, frontend offline queue'su veri tutarsızlığı yaratır. Backend agent'ın bulgu listesinde bu yok ama Bulgu #8'in backend ayağı olarak ele alınmalı.

---

### 6. TLS Bypass'ın Ortam Sınırı (Bulgu #12)

**Mimari değerlendirme:** `DangerousAcceptAnyServerCertificateValidator`'ın production'da açık olması bir konfigürasyon hatası ama asıl soru: **neden development'ta bile gerekli?**

Olası sebepler:
- Netsis self-signed sertifika kullanıyor
- ISS'in sertifika chain'i trusted root'ta değil
- Development ortamında local API HTTPS sertifikası

**Karar:** Backend agent'ın environment check çözümü acil düzeltme için yeterli. Ancak production'da Netsis veya ISS'in sertifikası trust edilmiyorsa, sadece environment check koymak o entegrasyonu kırar. Sertifika pinning veya explicit trust eklenmesi gerekebilir. Bu yayın öncesi test edilmeli.

## Critical Issues

### 1. Sıfır Test Coverage ile Bu Kapsamda Değişiklik Yapılması (Bulgu #7 — Tüm Diğerlerini Etkiler)

Tüm agent'lar bulguları doğru tespit etmiş ve düzeltme planları uygun. Ancak **hiçbir agent şu soruyu cevaplamıyor: bu düzeltmeler nasıl doğrulanacak?**

Mevcut durum: **sıfır test — backend ve frontend.** Önerilen değişiklikler:
- Authorization pipeline'ın generic constraint'i değişiyor (tüm MediatR pipeline etkileniyor)
- 3+ controller'a attribute ekleniyor
- Global fallback policy ekleniyor
- Netsis handler'ları rewrite ediliyor
- Password validation ekleniyor
- Frontend'e offline queue mekanizması ekleniyor

Bu değişikliklerin **herhangi birinde** regression olursa, tespit edecek **hiçbir mekanizma yok.** Bu tek başına reject sebebi.

**Zorunlu minimum test seti (deploy öncesi, pazarlık edilemez):**

| # | Test | Tipi | Doğruladığı Bulgu |
|---|---|---|---|
| T1 | Anonim istek → tüm endpoint'ler 401 döner | Integration | #2, #3 |
| T2 | Geçersiz token → 401 | Integration | #2 |
| T3 | Yetkisiz rol → 403 (Query + Command) | Integration | #3 |
| T4 | `[AllowAnonymous]` endpoint'ler (login, health) hâlâ çalışıyor | Integration | #2 |
| T5 | Boş/kısa/zayıf şifre → 400 validation error | Unit + Integration | #4 |

**Bu 9 test olmadan yapılan tüm düzeltmeler "güveniyoruz" statüsünde kalır. Production'da bunu kabul etmiyorum.**

### 2. Idempotency Backend Endpoint'i Tanımlı Değil (Architect C5 — Backend Agent'ta Eksik)

Architect doğru tespit etmiş: offline queue + retry = idempotent endpoint gerekli. Ancak:
- Backend agent'ın bulgu listesinde teslimat onayı endpoint'inin idempotent olması **yok**
- Frontend agent 409 handling yazmış ama backend'in 409 döneceğinin garantisi yok
- Mevcut endpoint duplicate confirmation'da ne yapıyor? İkinci kayıt mı oluşturuyor? 500 mü fırlatıyor? Bilinmiyor.

**Bu bir cross-layer gap. Frontend çözümü backend desteği olmadan tehlikeli.**

### 3. Pipeline Behavior Cascade Etkisi Doğrulanmamış (Architect C2)

`AuthorizationBehavior` constraint'i `ICommand<T>` → `IRequest<T>` olarak değişecek. Soru: başka hangi behavior'lar var ve bunlar ne yapıyor?

Backend agent bu soruyu cevaplamamış. Eğer:
- `TransactionBehavior` varsa ve `IRequest<T>` constraint'i varsa → tüm Query'ler transaction scope'a girer → **performans felaketi + deadlock riski**
- `ValidationBehavior` varsa ve Query'ler için validator yoksa → DI resolver'da `IValidator<T>` bulamama hatası olabilir (implementasyona bağlı)

**Bu kontrol yapılmadan auth düzeltmesi uygulanamaz.**

### 4. CariKod Alanının Mevcut Durumu Bilinmiyor (Bulgu #6)

Backend agent ve Architect "CariKod alanı yoksa migration gerekir" diyor. Ama:
- Customer entity'sinde bu alan **var mı, yok mu?** Bilinmiyor.
- Yoksa: migration + mevcut müşterilerin CariKod eşleştirmesi gerekli (veri hazırlığı)
- Varsa ama boş olanlar var mı? Kaç müşterinin CariKod'u tanımlı?

**Bu bilgi olmadan düzeltme planı "muhtemelen işe yarar" seviyesinde. Netsis'e yanlış fatura kesilmesi riski devam ediyor.**

### 5. Secret Rotation Doğrulaması Eksik

Tüm agent'lar "credential'ları rotate et" diyor. Doğru. Ama:
- Rotation sonrası **Netsis bağlantısı çalışıyor mu?** Test ediliyor mu?
- ISS import akışı rotation sonrası hâlâ çalışıyor mu?
- Google Maps API key değişikliği frontend'de hardcode mı, env variable mı?

Rotation yapılıp "çalışıyor mu" diye bakılmazsa, production'da entegrasyonlar kırılır. **Rotation + smoke test birlikte yapılmalı.**

### 6. Offline Queue'nun Sınır Durumları

Frontend agent'ın localStorage queue çözümü makul bir ilk adım. Ama doğrulanmamış edge case'ler:

- localStorage dolu ise (5MB limit) → `setItem` sessizce fail eder bazı tarayıcılarda → onay kaybolur
- Birden fazla tab açıksa → aynı queue'ya race condition ile yazabilir
- Telefon kapanır/browser crash → `flushOfflineQueue` hiç çağrılmaz, kullanıcı habersiz
- Queue'da 50+ item birikirse → hepsi sırayla gönderilir → rate limit veya timeout

**Bunlar P0 değil ama bilinmeli ve kabul edilmiş risk olarak dokümante edilmeli.**
